pagingdownload

Import files：

uigrow.paging.css
uigrow.paging.js

How to use：

write a div in html and put a id,then write a paging object,last use function uigrow_paging(pagingObject,id);

If you want ajax update content,you can build function uigrow_paging_customJumpToThePage(pagingObject,jumpPage),if have function name uigrow_paging_customJumpToThePage,it will use that as ajax entry.

Example：

<div id="div_paging"></div>

<script type="text/javascript">

paging();

function paging(){

var pagingObject={

currentPage:12,

totalPage:18,

totalRecords:81,

language:"English",

style:"simple",/*defalut not simple,you can try and look*/

/*//default，you can change this

dataCountTextFormer:"total",

dataCountTextLatter:"data",*/

hrefFormer:"test",

hrefLatter:".html"

};

var id="div_paging";

uigrow_paging(pagingObject,id);

}

/*//this use to ajax update

function uigrow_paging_customJumpToThePage(pagingObject,jumpPage){

pagingObject.currentPage=parseInt(jumpPage);

//Refresh paging

uigrow_paging(pagingObject,id);

}*/

</script>
